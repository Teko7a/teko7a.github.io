void solve() {
  std::string s;
  std::cin >> s;

  int n = s.size();
  int o = std::count(s.begin(), s.end(), '1');
  int a = o;

  for (int i = 0, j = -1, z = 0; i < n; ++i) {
    while (j < n - 1 && z < o) {
      j += 1;
      z += s[j] == '0';
      o -= s[j] != '0';
    }
    a = std::min(a, std::max(z, o));
    z -= s[i] == '0';
    o += s[i] != '0';
  }

  std::cout << a << '\n';
}
