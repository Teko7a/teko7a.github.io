void solve() {
  int n;
  std::string s;

  std::cin >> n >> s;

  int o = std::count(s.begin(), s.end(), '1');
  if (!o || o & 1) {
    return std::cout << "NO\n", void();
  }

  std::vector<std::pair<int, int>> ans, chain;

  for (int i = 0; i < n; ++i) {
    if (s[i] == '1') {
      int pos = i;
      while (s[(pos + 1) % n] == '0') {
        int nxt = (pos + 1) % n;
        ans.emplace_back(pos, nxt);
        pos = nxt;
      }
      chain.emplace_back(i, pos);
    }
  }

  for (int i = 1; i < (int) chain.size(); ++i) {
    ans.emplace_back(chain[0].second, chain[i].second);
  }

  std::cout << "YES\n";

  for (auto &[s, e] : ans) {
    std::cout << s + 1 << ' ' << e + 1 << '\n';
  }
}

void solve() {

  std::vector<int> p(n);

    std::cin >> i;
    --i;

  std::map<std::pair<int, int>, int> id;

  for (int i = 0, u, v; i < m; ++i) {
    std::cin >> u >> v;
    --u, --v;
    g[u].push_back(v);
    g[v].push_back(u);
    id[{u, v}] = id[{v, u}] = i + 1;
  }

  std::vector<int> ord(n, -1);

  for (int i = 0; i < n; ++i) {
    if (~ord[i]) {
      continue;
    }
    std::vector<int> a;

    int k = 0;
    for (int y = i; ord[y] == -1; y = p[y]) {
      ord[y] = k++;
      a.push_back(y);
    }
#define MOD(X) (X + k) % k
    for (auto &x : a) {
      std::sort(g[x].begin(), g[x].end(), [&](int i, int j) {
        return MOD(ord[i] - ord[x]) < MOD(ord[j] - ord[x]);
      });
    }
  }
